// Devolver una respuesta JSON devolviendo un objeto o una matriz
$app->get('/app-json-object', function() {
    $object = new \stdClass;
    $object->greeting = 'Hello World';
    return $object;
});

$app->get('/app-json-array', function() {
    return ['greeting' => 'Hello World'];
});
